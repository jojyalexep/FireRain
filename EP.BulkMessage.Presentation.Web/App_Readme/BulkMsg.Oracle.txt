CREATE TABLE CAMPAIGN
(
  ID               INTEGER,
  NAME             VARCHAR2(500 BYTE),
  TYPEID           INTEGER,
  STATUSDATE       DATE,
  CREATEDDATE      DATE,
  CREATEDBY        VARCHAR2(100 BYTE),
  STATUSID         INTEGER,
  CONTENTTEMPLATE  CLOB,
  SUBJECTTEMPLATE  VARCHAR2(500 BYTE),
  FILENAME         VARCHAR2(100 BYTE),
  PERCENTCOMPLETE  INTEGER,
  DEPARTMENTID     INTEGER,
  RECIPIENTFIELD   VARCHAR2(100 BYTE)
)
LOGGING 
NOCOMPRESS 
NOCACHE
NOPARALLEL
MONITORING;


CREATE UNIQUE INDEX CAMPAIGN_PK ON CAMPAIGN
(ID)
LOGGING
NOPARALLEL;


ALTER TABLE CAMPAIGN ADD (
  CONSTRAINT CAMPAIGN_PK
 PRIMARY KEY
 (ID));
